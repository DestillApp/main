<template>
  <!-- Teleport the modal backdrop to the body -->
  <teleport to="body">
    <!-- Modal backdrop -->
    <div class="modal-backdrop" @click.self="closeModal">
      <!-- Modal content transition -->
      <transition name="modal">
        <slot></slot>
      </transition>
    </div>
  </teleport>
</template>

<script>
import {  } from "vue";

/**
 * @component BaseModal
 * @description A base modal component that provides a backdrop for modal content.
 * @see closeModal
 * @emits close-modal
 */
export default {
  emits: ["close-modal"],
  setup(_, context) {
    /**
     * Closes the modal when the backdrop is clicked.
     * @function closeModal
     */
    const closeModal = () => {
      context.emit("close-modal");
    };

    return { closeModal };
  },
});
</script>
